import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();
        long mod = 1000000007;
        int a[] = new int[n + 1];
        long s[][] = new long[6][n + 1];
        for (int i = 1; i <= n; i++) {
            a[i] = sc.nextInt();
        }
        for (int i = 1; i <= n; i++) {
            int cur = 1;
            for (int k = 1; k <= 5; k++) {
                cur = (int) (cur * a[i] % mod);
                s[k][i] = (s[k][i - 1] + cur) % mod;
            }
        }
        for (int i = 1; i <= m; i++) {
            int l = sc.nextInt();
            int r = sc.nextInt();
            int k = sc.nextInt();
            long result = s[k][r] - s[k][l - 1];
            System.out.println(result);
        }
    }
}
